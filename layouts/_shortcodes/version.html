{{- /* Returns the current documentation full version name or the full name of the provided version. Usage: {{< version >}} -> 1.9.0 {{< version "2" >}} -> 2.0.0 */ -}} {{- $wantedVersion := .Get 0 -}} {{- $versions := partial "theme/functions/get-doc-versions.html" . -}} {{- $currentVersion := partial "theme/functions/get-doc-version.html" . -}} {{- $versionFullName := $currentVersion.full -}} {{- if $wantedVersion -}} {{- $version := where $versions "short" $wantedVersion -}} {{- $version = cond (gt (len $version) 0) (index $version 0) dict -}} {{- $versionFullName = $version.full -}} {{- end -}} {{- $versionFullName -}} {{- /* Strip trailing newline. */ -}}
